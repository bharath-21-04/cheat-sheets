Conversions

# Python
ord('A')       # 65
chr(65)        # 'A'

# JavaScript
'A'.charCodeAt(0)   // 65
String.fromCharCode(65)  // 'A'

# Quick math
'A' = 65    'a' = 97    '0' = 48
'a' - 'A' = 32  (case offset)
'0' to '9' = 48 to 57

# C
(int)'A'       // 65
(char)65      // 'A'
